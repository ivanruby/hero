hero.version

  • Type: String
  • Description: Returns the version of the library
  • Examples: 
    > hero.version
    // 0.5.0

hero.config()

  • Type: Method
  • Parameters: options (Object)
  • Description: Provides initial configuration for the library
  • Examples: 
    > hero.config({
  environment: 'dev' // or 'prod'
})
// 'dev' logs error while 'prod' doesn't

hero.log()

  • Type: Method
  • Parameters: msg (String), type (String)
  • Description: Logs information to the console about program behavior
  • Examples: 
    > hero.log("Hello World!", "info")
    // HERO SAYS: Hello World!
    
> hero.log("Hello World!", "warn")
    // HERO WARNS: Hello World! 
    
> hero.log("Hello World!", "error")
    // HERO HAD AN ERROR: Hello World!

hero.db.register()

  • Type: Method
  • Parameters: table_name (String)
  • Description: Adds a table to the hero database so that it can be used. Only registered tables can be accessed.
  • Examples: 
    > hero.db.register(table)

> hero.db.register('test')

hero.db.on()

  • Type: Method
  • Chainable: Yes
  • Parameters: table_name (String)
  • Description: Equivalent to SQL FROM command. Selects a table to be operated.
  • Examples: 
    > hero.db.on(table)
// Selects table 'test-table' to be operated.

> hero.db.on('test')
// 'test' is the name of the table

OR

> table = 'test';
hero.db.on(table)

hero.db.from()

  • Type: Method
  • Chainable: Yes
  • Parameters: table_name (String)
  • Description: Alias of hero.db.on()
    Equivalent to SQL FROM command. Selects a table to be operated.
  • Examples: 
    > hero.db.from(table)
// Selects table to be operated

> hero.db.from('test')

hero.db.get()

  • Type: Method
  • Chainable: No
  • Parameters: table_name (String) [OPTIONAL]
  • Description: Equivalent to SQL SELECT command. Returns an array of records in the given table. If the table is empty it returns an empty array
  • Examples: 
    > hero.db.get(table)
// []

> hero.db.get('test')
// []

> hero.db.on('test').get()
// []

hero.db.ok()

  • Type: Method
  • Chainable: No
  • Parameters: [NONE]
  • Description: Alias of hero.db.get()
    Equivalent to SQL SELECT command. Returns an array of records in the given table. If the table is empty it returns an empty array
  • Examples: 
    > hero.db.on(table).ok()
// []

hero.db.insert()

  • Type: Method
  • Chainable: Yes
  • Parameters: data (Object)
  • Description: Equivalent to SQL INSERT command. Inserts a row, represented by an object, in the table.
  • Examples: 
    > hero.db.on(table).insert({
  column1: value,
  column2: value,
        ...
  columnN: value
}).ok()
// true is successful, false if not

hero.db.on('test-table').insert({
  name: 'foo',
  age: 99
}).ok()

hero.db.update()

  • Type: Method
  • Chainable: Yes
  • Parameters: data (Object)
  • Description: Equivalent to SQL UPDATE command. Updates a row present in the table identified by the where condition
  • Examples: 
    > hero.db.on(table).update({
  column: newValue,
}).where({
  column: oldValue
}).ok()
// Returns true if successful, false if not

> hero.db.on('test').update({
  name: 'Foo Bar'
}).where({
  name: 'foo'
}).ok()

OR

> hero.db.update({
  name: 'Foo Bar'
}).on('test').where({
  name: 'foo'
}).ok()

OR

> hero.db.update({
  name: 'Foo Bar'
}).where({
  name: 'foo'
}).on('test').ok()

hero.db.remove()

  • Type: Method
  • Chainable: Yes
  • Parameters: condition (Object)
  • Description: Equivalent to SQL DELETE command. Deletes a row from the table, given by the condition.
    As of version 0.5.0 only equality conditions are supported.
  • Examples: 
    > hero.db.on(table).remove({
  column: value
}).ok()
// Returns true if successful, false if not

> hero.db.on('test-table').remove({
  name: 'foo'
}).ok()

OR

> hero.db.remove({
  name: 'foo'
}).from('test-table').ok()

hero.db.truncate()

  • Type: Method
  • Chainable: Yes
  • Parameters: table (String)
  • Description: Equivalent to SQL TRUNCATE command. Clears the whole table
  • Examples: 
    > hero.db.truncate(table).ok()
// Returns true if successful, false if not
// Depending on env 

> hero.db.truncate('test').ok()
// HERO SAYS: Table test truncated
// true

hero.db.sort()

  • Type: Method
  • Chainable: Yes
  • Parameters: column/property (String)
  • Description: Sorts a given table in ascending order according to the column/property given as argument
  • Examples: 
    > hero.db.on(table).sort().
// Returns an array of objects that represent the table rows if successful, an empty array if not

// Data in table 'test': [{name: 'foo', age: 25}, {name: 'bar', age: 50}]
> hero.db.on('test').sort('name').ok()
// [{name: 'bar', age: 50}, {name: 'foo', age: 25}]

> hero.db.on('test').sort('age').ok()
// [{name: 'foo', age: 25}, {name: 'bar', age: 50}]

hero.db.search()

  • Type: Method
  • Chainable: Yes
  • Parameters: data (String)
  • Description: Looks for a given key-value pair and returns true if found, false if not
  • Examples: 
    > hero.db.on(table).search(condition).ok()
// Returns true if successful, false if not

> hero.db.on('test').search({name: 'foo'}).ok()
// true

> hero.db.on('test').search({age: 100}).ok()
// false

hero.geo.isEnabled()

  • Type: Method
  • Chainable: No
  • Parameters: [NONE]
  • Description: Checks if the browser supports the Geolocation API
  • Examples: 
    > hero.geo.isEnabled()
// Returns true if enabled, false if not

hero.geo.init()

  • Type: Method
  • Chainable: No
  • Parameters: callback (Function)
  • Description: Checks if the browser supports the geolocation. If supported it executes an AJAX call to a default (metwit) api or a user-defined through the hero.config object and sends the result of the call as a parameter to the callback function
  • Examples: 
    > hero.geo.init(callback)
// Returns data if successful, null if not

> hero.geo.init(function(data) {
  if (data){
    // Consume the data
    console.log(data);
  }   
  else {
    // Handle error
    console.log("There was an error with the api call");
  }
});

OR

callback = function(data) {
  if (data){
    // Consume the data    
  }   
  else {
    // Handle error
  }
}

> hero.geo.init(callback);

hero.cache.create()

  • Type: Method
  • Chainable: Yes
  • Parameters: name (String)
  • Description: Specifies the name of the manifest file to be created

hero.cache.require()

  • Type: Method
  • Chainable: Yes
  • Parameters: files (String or Array of Strings)
  • Description: Specifies the the URLs of the files to be required

hero.cache.network()

  • Type: Method
  • Chainable: Yes
  • Parameters: files (String or Array of Strings)
  • Description: Specifies the URLs of the files that require network connectivity

hero.cache.fallback()

  • Type: Method
  • Chainable: Yes
  • Parameters: files (String or Array of Strings)
  • Description: Specifies the URLs of the files to be used as fallback when no network connectivity is available

hero.cache.generate()

  • Type: Method
  • Chainable: Yes
  • Parameters: [NONE]
  • Description: Generates the manifesto. As of version 0.5 it only outputs the manifesto file to the console.
Complete example for hero.cache 
// To be done on the server side
  hero.cache.create('test_app').require([
  '/index.html',
  '/css/style.css',
  'img/sprites.png'
]).network([
  '/login.html',
  '/register.html'
]).fallback('/no_network.html').generate()


/* Outputs on the console
CACHE MANIFEST
/login.html
/register.html

NETWORK:
/login.html
/register.html

FALLBACK:
/no_network.html 
*/
